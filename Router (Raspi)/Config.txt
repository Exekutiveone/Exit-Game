
========================================================================

✅ Temporäre statische IP setzen (bis zum Reboot)

	sudo ip addr add 192.168.4.1/24 dev wlan0

Falls nötig, die alte IP vorher löschen:

	sudo ip addr flush dev wlan0

========================================================================


========================================================================
✅ Konfiguration prüfen
/etc/hostapd/hostapd.conf
------------------------------------------------------------------------
	interface=wlan0
	ssid=espNet
	channel=6
	hw_mode=g
	auth_algs=1
	wpa=2
	wpa_passphrase=esp12345
	wpa_key_mgmt=WPA-PSK
	rsn_pairwise=CCMP
------------------------------------------------------------------------
/etc/default/hostapd
------------------------------------------------------------------------
	DAEMON_CONF="/etc/hostapd/hostapd.conf"
------------------------------------------------------------------------
========================================================================


========================================================================
✅ dnsmasq konfigurieren
	/etc/dnsmasq.conf (Minimalversion):
------------------------------------------------------------------------
interface=wlan0
	dhcp-range=192.168.4.10,192.168.4.100,255.255.255.0,24h
------------------------------------------------------------------------
========================================================================


========================================================================
✅ wpa_supplicant deaktivieren
------------------------------------------------------------------------
	sudo systemctl stop wpa_supplicant
	sudo systemctl disable wpa_supplicant
	sudo systemctl mask wpa_supplicant
------------------------------------------------------------------------
========================================================================


========================================================================
✅ Dienste starten
------------------------------------------------------------------------
	sudo systemctl restart hostapd
	sudo systemctl restart dnsmasq
------------------------------------------------------------------------
========================================================================


========================================================================
✅ Automatisch beim Boot starten
------------------------------------------------------------------------
	udo systemctl enable hostapd
	sudo systemctl enable dnsmasq
------------------------------------------------------------------------
========================================================================


========================================================================
✅ WLAN sichtbar
------------------------------------------------------------------------
	sudo systemctl status hostapd
------------------------------------------------------------------------
